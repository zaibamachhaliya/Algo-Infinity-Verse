Algorithm — C++

// Non-Preemptive Priority Scheduling (C++)
// Convention: lower priority number = higher priority

struct Process {
    int pid, arrival, burst, priority;
    int completion, turnaround, waiting;
};

void priorityScheduling(vector<Process>& procs)
{
    int n    = procs.size();
    int time = 0, done = 0;
    vector<bool> completed(n, false);

    while (done < n)
    {
        int idx = -1;

        // Find highest-priority process that has arrived
        for (int i = 0; i < n; i++)
        {
            if (!completed[i] && procs[i].arrival <= time)
            {
                if (idx == -1 ||
                    procs[i].priority < procs[idx].priority)
                    idx = i;
            }
        }

        if (idx == -1)          // no process arrived yet
        {
            time++;             // CPU idle — advance clock
            continue;
        }

        // Run to FULL completion (NON-PREEMPTIVE)
        time                    += procs[idx].burst;
        procs[idx].completion    = time;
        procs[idx].turnaround    = time - procs[idx].arrival;
        procs[idx].waiting       = procs[idx].turnaround
                                 - procs[idx].burst;
        completed[idx] = true;
        done++;
    }
}

// Metrics
// Turnaround Time (TAT) = Completion Time − Arrival Time
// Waiting Time    (WT)  = TAT − Burst Time
