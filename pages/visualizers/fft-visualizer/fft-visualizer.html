function FFT(P):
  n = length(P)
  if n == 1: return P
  
  Pe, Po = split(P, even), split(P, odd)
  ye, yo = FFT(Pe), FFT(Po)
  
  y = array of size n
  omega = e^(2 * PI * i / n)
  
  for j = 0 to n/2 - 1:
    y[j] = ye[j] + omega^j * yo[j]
    y[j + n/2] = ye[j] - omega^j * yo[j]
    
  return y
